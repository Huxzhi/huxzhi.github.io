                                                                                                                            # ref

                                                                                                                            з»‘е®ҡз®ҖеҚ•ж•°жҚ®зұ»еһӢ жҺҘеҸ—дёҖдёӘеҶ…йғЁеҖје№¶иҝ”еӣһдёҖдёӘ е“Қеә”ејҸ дё”еҸҜеҸҳзҡ„ ref еҜ№иұЎгҖӮref еҜ№иұЎд»…жңүдёҖдёӘ В .valueВ propertyпјҢжҢҮеҗ‘иҜҘеҶ…йғЁеҖјгҖӮ

                                                                                                                            з»‘е®ҡеӨҚжқӮзҡ„ж•°жҚ®зұ»еһӢ 06-Reactive е…Ёе®¶жЎ¶

                                                                                                                            жЎҲдҫӢ

                                                                                                                            жҲ‘д»¬иҝҷж ·ж“ҚдҪңжҳҜж— жі•ж”№еҸҳ messageВ  зҡ„еҖј еә”дёә message дёҚжҳҜе“Қеә”ејҸзҡ„ж— жі•иў« vue и·ҹиёӘиҰҒж”№жҲҗ ref

                                                                                                                            ж”№дёә ref

                                                                                                                            Ref TS еҜ№еә”зҡ„жҺҘеҸЈ

                                                                                                                            interface Ref<T> {
  value: T
}

                                                                                                                            жіЁж„Ҹиў« ref еҢ…иЈ…д№ӢеҗҺйңҖиҰҒ .value жқҘиҝӣиЎҢиөӢеҖј

                                                                                                                            # isRef

                                                                                                                            еҲӨж–ӯжҳҜдёҚжҳҜдёҖдёӘrefеҜ№иұЎ

                                                                                                                            import { ref, Ref,isRef } from 'vue'
let message: Ref<string | number> = ref("жҲ‘жҳҜmessage")
let notRef:number = 123
const changeMsg = () => {
  message.value = "change msg"
  console.log(isRef(message)); //true
  console.log(isRef(notRef)); //false

}

                                                                                                                            # customRef

                                                                                                                            иҮӘе®ҡд№ү ref

                                                                                                                            customRef жҳҜдёӘе·ҘеҺӮеҮҪж•°иҰҒжұӮжҲ‘д»¬иҝ”еӣһдёҖдёӘеҜ№иұЎ е№¶дё”е®һзҺ° get е’Ң setВ  йҖӮеҗҲеҺ»еҒҡйҳІжҠ–д№Ӣзұ»

                                                                                                                            <template>

  <div ref="div">е°Ҹж»ЎRef</div>
  <hr>
  <div>
    {{ name }}
  </div>
  <hr>
  <button @click="change">дҝ®ж”№ customRef</button>

</template>

<script setup lang='ts'>
import { ref, reactive, onMounted, shallowRef, customRef } from 'vue'

function myRef<T = any>(value: T) {
  let timer:any;
  return customRef((track, trigger) => {
    return {
      get() {
        track() //ж”¶йӣҶдҫқиө–
        return value
      },
      set(newVal) {
        clearTimeout(timer) //еҸҜд»Ҙи®ҫзҪ®йҳІжҠ–пјҢ500жҜ«з§’и§ҰеҸ‘дёҖж¬Ў
        timer =  setTimeout(() => {
          console.log('и§ҰеҸ‘дәҶset')
          value = newVal
          trigger() //и§ҰеҸ‘дҫқиө–
        },500)
      }
    }
  })
}


const name = myRef<string>('е°Ҹж»Ў')


const change = () => {
  name.value = 'еӨ§ж»Ў'
}

</script>
