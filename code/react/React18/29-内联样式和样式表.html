                                                                                                                                # 29-еҶ…иҒ”ж ·ејҸе’Ңж ·ејҸиЎЁ

                                                                                                                                React дёӯйҖҡиҝҮ import зӣҙжҺҘеј•е…ҘеӨ–йғЁзҡ„ CSS ж ·ејҸиЎЁпјҢдҪҶиҝҷе№¶дёҚжҳҜ React дёӯдҪҝз”Ё CSS зҡ„е”ҜдёҖж–№ејҸпјҢиҝҷйҮҢжҲ‘д»¬з®ҖеҚ•зҡ„зҪ—еҲ—еҮ з§Қ React дёӯи®ҫзҪ®ж ·ејҸзҡ„ж–№ејҸгҖӮ

                                                                                                                                # еҶ…иҒ”ж ·ејҸ

                                                                                                                                еңЁ React дёӯеҸҜд»ҘзӣҙжҺҘйҖҡиҝҮж Үзӯҫзҡ„ style еұһжҖ§жқҘдёәе…ғзҙ и®ҫзҪ®ж ·ејҸгҖӮstyle еұһжҖ§йңҖиҰҒзҡ„жҳҜдёҖдёӘеҜ№иұЎдҪңдёәеҖјпјҢжқҘдёәе…ғзҙ и®ҫзҪ®ж ·ејҸгҖӮ

                                                                                                                                <div style={{color:'red'}}>
    жҲ‘жҳҜDiv
</div>

                                                                                                                                дј йҖ’ж ·ејҸж—¶пјҢйңҖиҰҒжіЁж„ҸеҰӮжһңж ·ејҸеҗҚдёҚз¬ҰеҗҲй©јеі°е‘ҪеҗҚжі•пјҢйңҖиҰҒе°Ҷе…¶дҝ®ж”№дёәз¬ҰеҗҲй©јеі°е‘ҪеҗҚжі•зҡ„еҗҚеӯ—гҖӮжҜ”еҰӮпјҡbackground-color ж”№дёә backgroundColorгҖӮ

                                                                                                                                # еңЁеҶ…иҒ”ж ·ејҸдёӯдҪҝз”Ё State еҠЁжҖҒи®ҫзҪ®ж ·ејҸеҜ№иұЎ

                                                                                                                                еҰӮжһңеҶ…иҒ”ж ·ејҸзј–еҶҷиҝҮеӨҡпјҢдјҡеҜјиҮҙ JSX еҸҳеҫ—ејӮеёёж··д№ұпјҢжӯӨж—¶д№ҹеҸҜд»Ҙе°Ҷж ·ејҸеҜ№иұЎе®ҡд№үеҲ° JSX еӨ–пјҢз„¶еҗҺйҖҡиҝҮеҸҳйҮҸеј•е…ҘгҖӮ

                                                                                                                                и®ҫзҪ®ж ·ејҸж—¶пјҢеҸҜд»Ҙж №жҚ®дёҚеҗҢзҡ„ state еҖјеә”з”ЁдёҚеҗҢзҡ„ж ·ејҸпјҢжҜ”еҰӮжҲ‘д»¬еҸҜд»ҘеңЁз»„д»¶дёӯж·»еҠ дёҖдёӘжҢүй’®пјҢе№¶еёҢжңӣйҖҡиҝҮзӮ№еҮ»жҢүй’®еҸҜд»ҘеҲҮжҚў div зҡ„иҫ№жЎҶпјҢд»Јз ҒеҸҜд»Ҙиҝҷж ·еҶҷпјҡ

                                                                                                                                import React, {useState} from 'react';
import './App.css';

const App = () => {

    const [redBorder, setRedBorder] = useState(true);

    const pStyle = {
          color: 'red',
          backgroundColor: '#bfa',
          border: redBorder ? "red solid 1px" : "blue solid 1px"
    };

    const clickHandler = () => {
        setRedBorder(false);
    };

    return (
        <div>
            <p className={pStyle}>жҲ‘жҳҜдёҖдёӘж®өиҗҪ</p>
            <button onClick={clickHandler}>зӮ№жҲ‘дёҖдёӢ</button>
        </div>
    );
};

export default App;

                                                                                                                                StyleDemo.js

                                                                                                                                import React, {useState} from 'react';
import './StyleDemo.css';

const StyleDemo = () => {

    const [showBorder, setShowBorder] = useState(false);

    const toggleBorderHandler = ()=> {
      setShowBorder(prevState => !prevState);
    };

    return (
        <div className={`myDiv ${showBorder?' redBorder':''}`}>
            жҲ‘жҳҜDiv
            <button onClick={toggleBorderHandler}>еҲҮжҚўиҫ№жЎҶ</button>
        </div>
    );
};

export default StyleDemo;

                                                                                                                                # CSS Module

                                                                                                                                еҰӮжһңжІЎжңүзұ»еҗҚеҶІзӘҒзҡ„й—®йўҳпјҢеӨ–йғЁ CSS ж ·ејҸиЎЁдёҚеӨұдёәжҳҜдёҖз§ҚйқһеёёеҘҪзҡ„зј–еҶҷж ·ејҸзҡ„ж–№ејҸгҖӮдёәдәҶи§ЈеҶіиҝҷдёӘй—®йўҳ React дёӯиҝҳдёәжҲ‘д»¬жҸҗдҫӣдәҶдёҖдёӯж–№ејҸпјҢCSS ModuleгҖӮ

                                                                                                                                жҲ‘д»¬еҸҜд»Ҙе°Ҷ CSS Module зҗҶи§ЈдёәеӨ–йғЁж ·ејҸиЎЁзҡ„дёҖз§ҚиҝӣеҢ–зүҲпјҢе®ғзҡ„еӨ§йғЁеҲҶдҪҝз”Ёж–№ејҸйғҪе’ҢеӨ–йғЁж ·ејҸиЎЁзұ»дјјпјҢдёҚеҗҢзӮ№еңЁдәҺдҪҝз”Ё CSS Module еҗҺпјҢзҪ‘йЎөдёӯе…ғзҙ зҡ„зұ»еҗҚдјҡиҮӘеҠЁи®Ўз®—з”ҹжҲҗе№¶зЎ®дҝқе”ҜдёҖпјҢжүҖд»ҘдҪҝз”Ё CSS Module еҗҺпјҢжҲ‘д»¬еҶҚд№ҹдёҚз”ЁжӢ…еҝғзұ»еҗҚйҮҚеӨҚдәҶпјҒ

                                                                                                                                CSS Module еңЁ React дёӯе·Із»Ҹй»ҳи®Өж”ҜжҢҒдәҶпјҲеүҚжҸҗжҳҜдҪҝз”ЁдәҶ react-scriptsпјүпјҢжүҖд»Ҙж— йңҖеҶҚеј•е…Ҙе…¶д»–еӨҡдҪҷзҡ„жЁЎеқ—гҖӮдҪҝз”Ё CSS Module ж—¶йңҖиҰҒйҒөеҫӘеҰӮдёӢеҮ дёӘжӯҘйӘӨпјҡ

                                                                                                                                1. дҪҝз”Ё CSS Module зј–еҶҷзҡ„ж ·ејҸж–Үд»¶зҡ„ж–Үд»¶еҗҚеҝ…йЎ»дёәxxx.module.css
                                                                                                                                2. еңЁз»„д»¶дёӯеј•е…Ҙж ·ејҸзҡ„ж јејҸдёәimport xxx from './xxx.module.css'
                                                                                                                                3. и®ҫзҪ®зұ»еҗҚж—¶йңҖиҰҒдҪҝз”Ёxxx.yyyзҡ„еҪўејҸжқҘи®ҫзҪ®

                                                                                                                                иҜ·зңӢжЎҲдҫӢпјҡ

                                                                                                                                
/*

StyleDemo.module.css

*/
.myDiv{
    color: red;
    background-color: #bfa;
    font-size: 20px;
    border-radius: 12px;
}

                                                                                                                                /*

StyleDemo.js

*/
import styles from './StyleDemo.module.css';

const StyleDemo = () => {
    return (
        <div className={styles.myDiv}>
            жҲ‘жҳҜDiv
        </div>
    );
};

export default StyleDemo;

                                                                                                                                иҝҷе°ұжҳҜдёҖдёӘз®ҖеҚ•зҡ„ CSS Module зҡ„жЎҲдҫӢпјҢи®ҫзҪ®е®ҢжҲҗеҗҺдҪ еҸҜд»ҘиҮӘе·ұйҖҡиҝҮејҖеҸ‘иҖ…е·Ҙе…·жҹҘзңӢе…ғзҙ зҡ„ class еұһжҖ§пјҢдҪ дјҡеҸ‘зҺ° class еұһжҖ§е’ҢдҪ и®ҫзҪ®зҡ„е№¶дёҚе®Ңе…ЁдёҖж ·пјҢиҝҷжҳҜеӣ дёә CSS Module йҖҡиҝҮз®—жі•зЎ®дҝқдәҶжҜҸдёҖдёӘжЁЎеқ—дёӯзұ»еҗҚзҡ„е”ҜдёҖжҖ§гҖӮ

                                                                                                                                жҖ»д№ӢпјҢзӣёиҫғдәҺж ҮеҮҶзҡ„еӨ–йғЁж ·ејҸиЎЁжқҘиҜҙпјҢCSS Module е°ұжҳҜеӨҡдәҶдёҖзӮ№вҖ”вҖ”зЎ®дҝқзұ»еҗҚзҡ„е”ҜдёҖпјҢйҖҡиҝҮеҶ…йғЁз®—жі•йҒҝе…ҚдәҶдёӨдёӘз»„д»¶дёӯеҮәзҺ°йҮҚеӨҚзҡ„зұ»еҗҚпјҢеҰӮжһңдҪ иғҪдҝқиҜҒдёҚдјҡеҮәзҺ°йҮҚеӨҚзҡ„зұ»еҗҚпјҢе…¶е®һзӣҙжҺҘдҪҝз”ЁеӨ–йғЁж ·ејҸиЎЁд№ҹжҳҜдёҖж ·зҡ„гҖӮ
